93 std::pair<double,double> find_point (std::pair<double, double> marker_location, std::pair<double, double> explorer_location, double tolerance){
94  double dist1 = sqrt(sq(marker_location.first-explorer_location.first)+sq(marker_location.second-explorer_location.second));
95  double ratio = tolerance/dist1;
96  if (ratio < 0){
97  ratio = ratio*(-1);
98  }
99  else if (ratio > 1){
100  ratio = 1/ratio;
101  }
102  else {
103  ratio = ratio;
104  }
105  std::pair<double, double> follower_location;
106  follower_location.first = (((1-ratio)*explorer_location.first)+(ratio*marker_location.first));
107  follower_location.second = (((1-ratio)*explorer_location.second)+(ratio*marker_location.second));
108  return follower_location;
109 }
